class CreatingPageView(MethodView):
    def post(self):
        try:
            first_name = request.form.get('first_name')
            last_name = request.form.get('last_name')
            gender = request.form.get('gender')
            date_of_birth = request.form.get('date_of_birth')
            date_of_death = request.form.get('date_of_death')
            place_of_birth = request.form.get('place_of_birth')
            nationality = request.form.get('nationality')
            life_profile = request.form.get('life_profile')
            portrait = request.form.get('portrait')
            theme = request.form.get('theme')
            creating_date = 'dsa'

            pageInfo = DeceasedPage(first_name, last_name, gender, date_of_birth, date_of_death, \
                                    place_of_birth, nationality, life_profile, portrait, theme, creating_date)
            db.session.add(pageInfo)
            db.session.commit()
            response = {
                'message': 'You created successfully.',
                'first_name': pageInfo.first_name,
                'last_name': last_name,
                'gender': gender,
                'date_of_birth': date_of_birth,
                'date_of_death': date_of_death,
                'place_of_birth': place_of_birth,
                'nationality': nationality,
                'life_profile': life_profile,
                'portrait': portrait,
                'theme': theme
            }
            return make_response(jsonify(response)), 200

        except Exception as e:
            # Create a response containing an string error message
            response = {
                'message': str(e) + '1'
            }
            # Return a server error using the HTTP Error Code 500 (Internal Server Error)
            return make_response(jsonify(response)), 500
